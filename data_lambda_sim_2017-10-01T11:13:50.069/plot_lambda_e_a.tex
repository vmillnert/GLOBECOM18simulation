\documentclass[tikz]{standalone}
\usepackage{pgfplots}

\pgfplotsset{compat = 1.10}

% load the path-wise input data
\pgfplotstableread[col sep = tab]{e_a.txt}\ea
\pgfplotstableread[col sep = tab]{u_mean.txt}\umean

\begin{document}
\begin{tikzpicture}
  \begin{axis}[
    width = 4cm,
    height = 3cm,
    scale only axis,
    xmin = 0, xmax = 1,
    legend cell align =left,
    legend style = {at={(0.80, 0.05)}, anchor=south east},
    ymin = 0.9, ymax = 1.00,
    ylabel = utility,
    xlabel = lambda,
    ytick = {0.9, 0.95, 1},
    % legend pos = south east,
    legend columns = 3
    ]

    \addplot[only marks,  green!50] table[x index = {0}, y index = {1}]{\ea}; \label{e}
    \addplot[only marks,  orange!50] table[x index = {0}, y index = {2}]{\ea}; \label{a}
    \addplot[only marks, mark=o, blue!50] table[x index = {0}, y index = {1}]{\umean}; \label{u}

    \addlegendentry{$u^\mathrm{e}$}
    \addlegendentry{$u^\mathrm{a}$}
    \addlegendentry{$U$}

  \end{axis}

\end{tikzpicture}
\end{document}
